实现上传功能

2024年4月7日 · 阅读需 4 分钟
moubing
rookie devloper

这是一个作业，要求是通过http协议传输图片，客户端要读取图片并将其转换为字节流或者Base64编码，在发送给服务端，目前我就是使用一个input来选择上传的图片，并没有将它转换，服务端接收到后使用formidable来解析post请求体，然后将图片保存在静态资源目录public/image目录中，并将图片的名称存入了本地的mongodb数据库。

const border = document.querySelector('.border')
const imgInput = document.querySelector('.imgInput')
const imgList = document.querySelector('.imgList')
const uploadInput = document.querySelector('form input')

const imgArray = []

border.addEventListener('click', () => {
    imgInput.click()
})

imgInput.addEventListener('change', updateImgList)

function updateImgList() {
    while(imgList.firstChild) {
        imgList.remove(imgList.firstChild)
    }

    const files = imgInput.files
    if(files.length === 0) {
        const listItem = document.createElement('p')
        listItem.innerText = "你并没有上传图片"
        imgList.append(listItem)
    } else {
        
        for(const file of files) {
            if(validFile(file.type)){
                imgArray.push(file)
                const newImgItem = document.createElement('div')
                const fileName = document.createElement('div')
                const fileSize = document.createElement('div')
                const img = document.createElement('img')
                fileName.innerText = file.name
                fileSize.innerText = computeSize(file.size)
                img.src = URL.createObjectURL(file)
                img.alt = file.name
                fileName.classList.add('file-name')
                fileSize.classList.add('file-size')
                img.classList.add('imgItem')
                newImgItem.append(fileName)
                newImgItem.append(fileSize)
                newImgItem.append(img)
                imgList.append(newImgItem)

            } else {
                const listItem = document.createElement('p')
                listItem.innerText = `你上传文件${file.name}不是图片格式，它是${file.type}格式的`
                imgList.append(listItem)
            }
        }
    }
const finalObj = imgArray.reduce((obj,curfile) => {
    obj[curfile.name] = curfile
    return obj
}, {})

function computeSize(number) {
    if (number < 1024) {
        return `${number} bytes`;
      } else if (number >= 1024 && number < 1048576) {
        return `${(number / 1024).toFixed(1)} KB`;
      } else if (number >= 1048576) {
        return `${(number / 1048576).toFixed(1)} MB`;
      }
}

function validFile(type) {
    return fileTypes.includes(type)
}

const fileTypes = [
    "image/apng",
    "image/bmp",
    "image/gif",
    "image/jpeg",
    "image/pjpeg",
    "image/png",
    "image/svg+xml",
    "image/tiff",
    "image/webp",
    "image/x-icon",
  ];
